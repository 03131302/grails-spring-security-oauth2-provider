001package grails.plugin.springsecurity.oauthprovider.provider;
002
003import org.springframework.security.core.authority.AuthorityUtils;
004import org.springframework.security.oauth2.common.exceptions.InvalidClientException;
005import org.springframework.security.oauth2.common.exceptions.InvalidGrantException;
006import org.springframework.security.oauth2.common.exceptions.InvalidScopeException;
007import org.springframework.security.oauth2.common.util.OAuth2Utils;
008import org.springframework.security.oauth2.provider.*;
009import org.springframework.security.oauth2.provider.request.DefaultOAuth2RequestFactory;
010
011import java.util.*;
012
013/**
014 * Based heavily on {@link DefaultOAuth2RequestFactory} which initializes fields from the parameters map, validates
015 * grant types and scopes, and fills in scopes with the default values from the client if they are missing.
016 *
017 * The API has undergone some dramatic changes compared to GrailsAuthorizationRequestManager. This stays close to
018 * the original, keeping the option scopeRequired. Some code duplication was necessary due to private methods.
019 *
020 * @author Dave Syer
021 * @author Amanda Anganes
022 * @author Roy Willemse
023 */
024
025public class GrailsOAuth2RequestFactory extends DefaultOAuth2RequestFactory {
026
027    private final ClientDetailsService clientDetailsService;
028
029    private SecurityContextAccessor securityContextAccessor = new DefaultSecurityContextAccessor();
030
031    private boolean checkUserScopes = false;
032
033    private final boolean scopeRequired;
034
035    public GrailsOAuth2RequestFactory(ClientDetailsService clientDetailsService, boolean scopeRequired) {
036        super(clientDetailsService);
037        this.clientDetailsService = clientDetailsService;
038        this.scopeRequired = scopeRequired;
039    }
040
041    @Override
042    public void setCheckUserScopes(boolean checkUserScopes) {
043        this.checkUserScopes = checkUserScopes;
044    }
045
046    @Override
047    public void setSecurityContextAccessor(SecurityContextAccessor securityContextAccessor) {
048        this.securityContextAccessor = securityContextAccessor;
049    }
050
051    @Override
052    public AuthorizationRequest createAuthorizationRequest(Map<String, String> authorizationParameters) {
053
054        String clientId = authorizationParameters.get(OAuth2Utils.CLIENT_ID);
055        String state = authorizationParameters.get(OAuth2Utils.STATE);
056        String redirectUri = authorizationParameters.get(OAuth2Utils.REDIRECT_URI);
057        Set<String> responseTypes = OAuth2Utils.parseParameterList(authorizationParameters
058                .get(OAuth2Utils.RESPONSE_TYPE));
059
060        Set<String> scopes = extractScopes(authorizationParameters, clientId);
061
062        AuthorizationRequest request = new AuthorizationRequest(authorizationParameters,
063                Collections.<String, String>emptyMap(), clientId, scopes, null, null, false, state, redirectUri,
064                responseTypes);
065
066        ClientDetails clientDetails = clientDetailsService.loadClientByClientId(clientId);
067        validateGrantTypes(clientDetails);
068
069        request.setResourceIdsAndAuthoritiesFromClientDetails(clientDetails);
070
071        return request;
072
073    }
074
075    @Override
076    public TokenRequest createTokenRequest(Map<String, String> requestParameters, ClientDetails authenticatedClient) {
077
078        String clientId = requestParameters.get(OAuth2Utils.CLIENT_ID);
079        if (clientId == null) {
080            // if the clientId wasn't passed in in the map, we add pull it from the authenticated client object
081            clientId = authenticatedClient.getClientId();
082        }
083        else {
084            // otherwise, make sure that they match
085            if (!clientId.equals(authenticatedClient.getClientId())) {
086                throw new InvalidClientException("Given client ID does not match authenticated client");
087            }
088        }
089        String grantType = requestParameters.get(OAuth2Utils.GRANT_TYPE);
090
091        Set<String> scopes = extractScopes(requestParameters, clientId);
092        TokenRequest tokenRequest = new TokenRequest(requestParameters, clientId, scopes, grantType);
093
094        ClientDetails clientDetails = clientDetailsService.loadClientByClientId(clientId);
095        validateGrantTypes(clientDetails);
096
097        return tokenRequest;
098    }
099
100    private Set<String> extractScopes(Map<String, String> requestParameters, String clientId) {
101        Set<String> scopes = OAuth2Utils.parseParameterList(requestParameters.get(OAuth2Utils.SCOPE));
102        ClientDetails clientDetails = clientDetailsService.loadClientByClientId(clientId);
103        boolean scopesNotPresent = scopesNotPresent(scopes);
104
105        // Use the client's registered defaults per spec if request isn't required to include scope
106        if(scopesNotPresent && !scopeRequired) {
107            scopes = clientDetails.getScope();
108        }
109        else if (scopesNotPresent && scopesCanBeOmitted(requestParameters)) {
110            scopes = Collections.emptySet();
111        }
112
113        if (checkUserScopes) {
114            scopes = checkUserScopes(scopes, clientDetails);
115        }
116        return scopes;
117    }
118
119    private Set<String> checkUserScopes(Set<String> scopes, ClientDetails clientDetails) {
120        if (!securityContextAccessor.isUser()) {
121            return scopes;
122        }
123        Set<String> result = new LinkedHashSet<String>();
124        Set<String> authorities = AuthorityUtils.authorityListToSet(securityContextAccessor.getAuthorities());
125        for (String scope : scopes) {
126            if (authorities.contains(scope) || authorities.contains(scope.toUpperCase())
127                    || authorities.contains("ROLE_" + scope.toUpperCase())) {
128                result.add(scope);
129            }
130        }
131        return result;
132    }
133
134    private boolean scopesNotPresent(Set<String> scopes) {
135        return (scopes == null) || (scopes.isEmpty());
136    }
137
138    // The access token request for the authorization code grant is the
139    // only case where the scope param is not mentioned in the spec
140    private boolean scopesCanBeOmitted(Map<String, String> requestParameters) {
141        String grantType = requestParameters.get("grant_type");
142        return (grantType != null) && (grantType.equals("authorization_code"));
143    }
144
145    private void validateGrantTypes(ClientDetails clientDetails) {
146        Collection<String> authorizedGrantTypes = clientDetails.getAuthorizedGrantTypes();
147        if (authorizedGrantTypes == null || authorizedGrantTypes.isEmpty()) {
148            throw new InvalidGrantException("A client must have at least one authorized grant type.");
149        }
150    }
151
152}
