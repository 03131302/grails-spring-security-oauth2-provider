001package grails.plugin.springsecurity.oauthprovider.provider;
002
003import org.springframework.security.oauth2.common.exceptions.InvalidClientException;
004import org.springframework.security.oauth2.common.exceptions.InvalidGrantException;
005import org.springframework.security.oauth2.common.exceptions.InvalidScopeException;
006import org.springframework.security.oauth2.common.util.OAuth2Utils;
007import org.springframework.security.oauth2.provider.*;
008
009import java.util.Collection;
010import java.util.Collections;
011import java.util.Map;
012import java.util.Set;
013
014public class GrailsAuthorizationRequestManager extends DefaultAuthorizationRequestManager {
015
016    private final ClientDetailsService clientDetailsService;
017    private final boolean scopeRequired;
018
019    public GrailsAuthorizationRequestManager(ClientDetailsService clientDetailsService, boolean scopeRequired) {
020        super(clientDetailsService);
021        this.clientDetailsService = clientDetailsService;
022        this.scopeRequired = scopeRequired;
023    }
024
025    public AuthorizationRequest createAuthorizationRequest(Map<String, String> parameters) {
026
027        String clientId = extractClientId(parameters);
028        ClientDetails clientDetails = clientDetailsService.loadClientByClientId(clientId);
029        validateGrantTypes(clientDetails);
030
031        Set<String> scopes = extractScopes(parameters);
032
033        DefaultAuthorizationRequest request = new DefaultAuthorizationRequest(parameters,
034                Collections.<String, String> emptyMap(), clientId, scopes);
035
036        request.addClientDetails(clientDetails);
037        return request;
038    }
039
040    private String extractClientId(Map<String, String> parameters) {
041        String clientId = parameters.get("client_id");
042        if (clientId == null) {
043            throw new InvalidClientException("A client id must be provided");
044        }
045        return clientId;
046    }
047
048    private Set<String> extractScopes(Map<String, String> parameters) {
049        Set<String> scopes = OAuth2Utils.parseParameterList(parameters.get("scope"));
050        boolean scopesNotPresent = scopesNotPresent(scopes);
051
052        if(scopesNotPresent && (scopesCanBeOmitted(parameters) || !scopeRequired)) {
053            scopes = Collections.emptySet();
054        }
055        else if(scopesNotPresent) {
056            throw new InvalidScopeException("Scope must be specified");
057        }
058        return scopes;
059    }
060
061    private boolean scopesNotPresent(Set<String> scopes) {
062        return (scopes == null) || (scopes.isEmpty());
063    }
064
065    // The access token request for the authorization code grant is the
066    // only case where the scope param is not mentioned in the spec
067    private boolean scopesCanBeOmitted(Map<String, String> parameters) {
068        String grantType = parameters.get("grant_type");
069        return (grantType != null) && (grantType.equals("authorization_code"));
070    }
071
072    private void validateGrantTypes(ClientDetails clientDetails) {
073        Collection<String> authorizedGrantTypes = clientDetails.getAuthorizedGrantTypes();
074        if(authorizedGrantTypes == null || authorizedGrantTypes.isEmpty()) {
075            throw new InvalidGrantException("A client must have at least one authorized grant type.");
076        }
077    }
078}
