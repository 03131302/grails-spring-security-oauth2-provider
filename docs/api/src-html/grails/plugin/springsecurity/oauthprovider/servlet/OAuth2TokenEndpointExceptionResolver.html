001package grails.plugin.springsecurity.oauthprovider.servlet;
002
003import grails.plugin.springsecurity.oauthprovider.exceptions.OAuth2TokenEndpointException;
004import org.springframework.http.ResponseEntity;
005import org.springframework.security.oauth2.common.exceptions.OAuth2Exception;
006import org.springframework.security.oauth2.provider.ClientRegistrationException;
007import org.springframework.security.oauth2.provider.endpoint.TokenEndpoint;
008import org.springframework.security.oauth2.provider.error.OAuth2ExceptionRenderer;
009import org.springframework.web.context.request.ServletWebRequest;
010import org.springframework.web.servlet.ModelAndView;
011import org.springframework.web.servlet.handler.AbstractHandlerExceptionResolver;
012
013import javax.servlet.http.HttpServletRequest;
014import javax.servlet.http.HttpServletResponse;
015
016public class OAuth2TokenEndpointExceptionResolver extends AbstractHandlerExceptionResolver {
017
018    private TokenEndpoint tokenEndpoint;
019    private OAuth2ExceptionRenderer exceptionRenderer;
020
021    public void setTokenEndpoint(TokenEndpoint tokenEndpoint) {
022        this.tokenEndpoint = tokenEndpoint;
023    }
024
025    public void setExceptionRenderer(OAuth2ExceptionRenderer exceptionRenderer) {
026        this.exceptionRenderer = exceptionRenderer;
027    }
028
029    @Override
030    protected ModelAndView doResolveException(HttpServletRequest request,
031            HttpServletResponse response, Object handler, Exception ex) {
032
033        logger.debug("Entering token endpoint exception resolver");
034
035        if(ex instanceof OAuth2TokenEndpointException) {
036            OAuth2TokenEndpointException oae = (OAuth2TokenEndpointException) ex;
037            return handleException(request, response, oae);
038        }
039        return null;
040    }
041
042    private ModelAndView handleException(HttpServletRequest request, HttpServletResponse response,
043                                         OAuth2TokenEndpointException tokenEndpointException) {
044        try {
045            logger.debug("Handling token endpoint exception: " + tokenEndpointException.getCause());
046            ResponseEntity<OAuth2Exception> entity = extractResponseEntity(tokenEndpointException);
047
048            ServletWebRequest webRequest = new ServletWebRequest(request, response);
049            exceptionRenderer.handleHttpEntityResponse(entity, webRequest);
050            return new ModelAndView();
051        }
052        catch (Exception e) {
053            throw new RuntimeException("An error occurred while resolving token endpoint exception", e);
054        }
055    }
056
057    private ResponseEntity<OAuth2Exception> extractResponseEntity(OAuth2TokenEndpointException e) throws Exception {
058        Throwable cause = e.getCause();
059
060        if(cause instanceof ClientRegistrationException) {
061            ClientRegistrationException cre = (ClientRegistrationException) cause;
062            return tokenEndpoint.handleClientRegistrationException(cre);
063        }
064        else {
065            OAuth2Exception oae = (OAuth2Exception) cause;
066            return tokenEndpoint.handleException(oae);
067        }
068    }
069}
