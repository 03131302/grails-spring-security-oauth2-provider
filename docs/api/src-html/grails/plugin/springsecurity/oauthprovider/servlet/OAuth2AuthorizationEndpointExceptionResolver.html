001package grails.plugin.springsecurity.oauthprovider.servlet;
002
003import grails.plugin.springsecurity.oauthprovider.exceptions.OAuth2AuthorizationEndpointException;
004import org.springframework.security.oauth2.common.exceptions.OAuth2Exception;
005import org.springframework.security.oauth2.provider.ClientRegistrationException;
006import org.springframework.security.oauth2.provider.endpoint.AuthorizationEndpoint;
007import org.springframework.web.context.request.ServletWebRequest;
008import org.springframework.web.servlet.ModelAndView;
009import org.springframework.web.servlet.handler.AbstractHandlerExceptionResolver;
010
011import javax.servlet.http.HttpServletRequest;
012import javax.servlet.http.HttpServletResponse;
013
014public class OAuth2AuthorizationEndpointExceptionResolver extends AbstractHandlerExceptionResolver {
015
016    private AuthorizationEndpoint authorizationEndpoint;
017
018    public void setAuthorizationEndpoint(AuthorizationEndpoint authorizationEndpoint) {
019        this.authorizationEndpoint = authorizationEndpoint;
020    }
021
022    @Override
023    protected ModelAndView doResolveException(HttpServletRequest request,
024                  HttpServletResponse response, Object handler, Exception ex) {
025
026        logger.debug("Entering authorization endpoint exception resolver");
027
028        if(ex instanceof OAuth2AuthorizationEndpointException) {
029            OAuth2AuthorizationEndpointException oae = (OAuth2AuthorizationEndpointException) ex;
030            return handleException(request, response, oae);
031        }
032
033        return null;
034    }
035
036    private ModelAndView handleException(HttpServletRequest request, HttpServletResponse response,
037                                         OAuth2AuthorizationEndpointException authorizationEndpointException) {
038        try {
039            Throwable cause = authorizationEndpointException.getCause();
040            ServletWebRequest webRequest = new ServletWebRequest(request, response);
041
042            logger.debug("Handling authorization endpoint exception: " + authorizationEndpointException.getCause());
043
044            if(isClientRegistration(cause)) {
045                ClientRegistrationException cre = (ClientRegistrationException) cause;
046                return authorizationEndpoint.handleClientRegistrationException(cre, webRequest);
047            }
048            else if(isOAuth2Exception(cause)) {
049                OAuth2Exception oae = (OAuth2Exception) cause;
050                return authorizationEndpoint.handleOAuth2Exception(oae, webRequest);
051            }
052            else {
053                throw new IllegalStateException("Invalid OAuthorizationEndpointException", cause);
054            }
055
056        }
057        catch (Exception e) {
058            throw new RuntimeException("An error occurred while resolving authorization endpoint exception", e);
059        }
060    }
061
062    private boolean isClientRegistration(Throwable cause) {
063        return cause instanceof ClientRegistrationException;
064    }
065
066    private boolean isOAuth2Exception(Throwable cause) {
067        return cause instanceof OAuth2Exception;
068    }
069}
